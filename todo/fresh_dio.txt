import 'package:dio/dio.dart';
import 'package:fresh_dio/fresh_dio.dart';

// Create a Dio instance for your API requests
final dio = Dio(
  BaseOptions(
    baseUrl: 'https://api.example.com',
  ),
);

// Define your token refresh logic
Future<OAuth2Token?> tokenRefresh(OAuth2Token? token, Dio dio) async {
  try {
    // Make a request to the server to get a new token using the refresh token
    final response = await dio.post(
      '/auth/refresh',
      data: {'refreshToken': token?.refreshToken},
    );
    // Return the new token
    return OAuth2Token.fromJson(response.data);
  } on DioException {
    // If the refresh fails (e.g., refresh token is also invalid),
    // throw a RevokeTokenException to force a full logout.
    throw const RevokeTokenException();
  }
}

// Create and add the Fresh interceptor to your Dio instance
final fresh = Fresh.oAuth2(
  token: null, // Initial token
  tokenRefresh: tokenRefresh,
);

dio.interceptors.add(fresh);
